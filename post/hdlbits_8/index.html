            q <= 0;

endmodule

module top_module (
    input clk,
    input reset,
    input [31:0] in,
    output reg [31:0] out
);
    reg [31:0] q;
    always @(posedge clk)
        q <= in;
    
    integer i;
    always @(posedge clk) begin
        if(reset)
           out <= 0;
        else begin
            for(i = 0; i < 32; i = i + 1) begin
                if(q[i] & ~in[i])
                    out[i] <= 1'b1;
            end
        end
    end
        
endmodule
